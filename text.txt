    <script type="module">
        // Importing Firebase modules using recommended stable versions
        import { initializeApp } from "https://www.gstatic.com/firebasejs/9.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/9.6.1/firebase-auth.js";
        import { getFirestore, setLogLevel } from "https://www.gstatic.com/firebasejs/9.6.1/firebase-firestore.js";

        // Global Firebase Configuration (Mandatory Variables)
        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {};
        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;

        // --- Firebase & Auth Setup ---
        let isAuthReady = false;

        if (Object.keys(firebaseConfig).length > 0) {
            const app = initializeApp(firebaseConfig);
            const auth = getAuth(app);
            setLogLevel('Debug');
            
            // Simplified Auth Initialization 
            (async () => {
                try {
                    if (initialAuthToken) {
                        await signInWithCustomToken(auth, initialAuthToken);
                    } else {
                        await signInAnonymously(auth);
                    }
                } catch (e) {
                    console.error("Firebase Auth failed, continuing with anonymous sign-in if possible.", e);
                    if (!auth.currentUser) await signInAnonymously(auth);
                }
                isAuthReady = true;
                const userInput = document.getElementById('user-input');
                const sendButton = document.getElementById('send-button');
                sendButton.disabled = userInput.value.trim() === '';
            })();
        } else {
            isAuthReady = true;
            document.getElementById('send-button').disabled = document.getElementById('user-input').value.trim() === '';
        }

        // --- UI Elements ---
        const chatForm = document.getElementById('chat-form');
        const userInput = document.getElementById('user-input');
        const chatContainer = document.getElementById('chat-container');
        const sendButton = document.getElementById('send-button');
        const loadingSpinner = document.getElementById('loading-spinner');
        const sendIcon = document.getElementById('send-icon');
        const qrButton = document.getElementById('qr-button');
        const qrModalOverlay = document.getElementById('qr-modal-overlay');
        const closeQrModal = document.getElementById('close-qr-modal');

        // --- Chat UI Helper Functions ---
        const addMessage = (text, sender) => {
            const messageWrapper = document.createElement('div');
            messageWrapper.classList.add('flex', 'flex-col', sender === 'user' ? 'items-end' : 'items-start');

            const messageBubble = document.createElement('div');
            messageBubble.classList.add('message-bubble', sender === 'user' ? 'user-message' : 'ai-message');

            const formattedText = text
                .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
                .replace(/\*(.*?)\*/g, '<em>$1</em>')
                .replace(/\n/g, '<br>');

            messageBubble.innerHTML = formattedText;
            messageWrapper.appendChild(messageBubble);
            chatContainer.appendChild(messageWrapper);

            chatContainer.scrollTop = chatContainer.scrollHeight;
        };

        const toggleLoading = (isLoading) => {
            userInput.disabled = isLoading;
            sendButton.disabled = isLoading || userInput.value.trim() === '';
            loadingSpinner.classList.toggle('hidden', !isLoading);
            sendIcon.classList.toggle('hidden', isLoading);
            sendButton.classList.toggle('bg-sky-600', !isLoading);
            sendButton.classList.toggle('bg-gray-400', isLoading);
        };

        // --- Gemini API Logic ---
        const systemInstruction = "أنت مساعد جراحة الوجه والفكين لما بعد العمليات اسمه 'راحتي'، تتمتع بمعرفة عالية وتعاطف كبير وتركيز على السلامة، وتعمل لصالح عيادة الدكتور مشاري الغامدي، أخصائي جراحة الفم والوجه والفكين المرخص من الهيئة السعودية للاختصاصات الصحية. هدفك الوحيد هو تقديم تعليمات واضحة وبسيطة وداعمة وإجابات على أسئلة المرضى حول تعافيهم. يجب أن تكون جميع ردودك باللغة العربية الفصحى الواضحة والقياسية. أعطِ الأولوية دائمًا لسلامة المريض. لأي أعراض خطيرة أو عاجلة (مثل ارتفاع درجة الحرارة فوق 38.5 درجة مئوية، نزيف أحمر فاتح لا يمكن السيطرة عليه، ألم شديد مفاجئ، أو صعوبة في التنفس/البلع)، يجب أن تنصح المستخدم بشكل صريح بالاتصال الفوري بمركز الاتصال الموحد على الرقم **٩٢٠٠٢٢٢٣١** أو خدمات الطوارئ. في حال استمرار الألم لأكثر من ثلاثة أيام دون أعراض خطيرة، انصح المريض بمراجعة عيادة **٣١٤** في الدور **السادس** من المبنى رقم **٢** بمجمع الدمام الطبي. يجب الإشارة إلى أن الدكتور مختص ومرخص من الهيئة السعودية للاختصاصات الصحية. غطِ مواضيع ما بعد الجراحة الرئيسية مثل النزيف، التورم، النظام الغذائي، ونظافة الفم. اجعل الإجابات موجزة واستخدم التغميق **bolding** للتأكيد على الكلمات المفتاحية.";
        
        const sendMessage = async (query) => {
            if (!query.trim() || !isAuthReady) return;

            addMessage(query, 'user');
            userInput.value = '';
            toggleLoading(true);

            const apiKey = "";
            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            const MAX_RETRIES = 3;
            let attempt = 0;

            while (attempt < MAX_RETRIES) {
                try {
                    const payload = {
                        contents: [{ parts: [{ text: query }] }],
                        tools: [{ "google_search": {} }],
                        systemInstruction: { parts: [{ text: systemInstruction }] },
                    };

                    const response = await fetch(apiUrl, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });

                    if (!response.ok) {
                        if (response.status === 429 && attempt < MAX_RETRIES - 1) {
                            const delay = Math.pow(2, attempt) * 1000 + Math.random() * 1000;
                            await new Promise(resolve => setTimeout(resolve, delay));
                            attempt++;
                            continue;
                        }
                        throw new Error(`API call failed with status: ${response.status}`);
                    }

                    const result = await response.json();
                    const text = result.candidates?.[0]?.content?.parts?.[0]?.text;

                    if (text) {
                        addMessage(text, 'ai');
                    } else {
                        throw new Error("AI response text is empty or malformed.");
                    }

                    break; // Success

                } catch (error) {
                    console.error("Gemini API Error:", error);
                    attempt++;

                    if (attempt === MAX_RETRIES) {
                        // Display final, professional error message after all retries fail
                        addMessage('عذراً، فشل الاتصال بخدمة المساعد الآلي بعد عدة محاولات. يرجى التأكد من اتصالك بشبكة الإنترنت القوية. **للحالات الطارئة، يرجى الاتصال فوراً بمركز الاتصال الموحد ٩٢٠٠٢٢٢٣١**.', 'ai');
                    } else {
                        const delay = Math.pow(2, attempt) * 1000 + Math.random() * 500;
                        await new Promise(resolve => setTimeout(resolve, delay));
                    }
                }
            }
            toggleLoading(false);
        };


        // --- QR Code Logic (Uses the self-contained qrcode function) ---
        
        // Final working URL for the QR code, based on user-provided GitHub path
        const qrDataToEncode = "https://mishari2003.github.io/Rahty-chatbot-clinic/"; 
        const qrcodeCanvasDiv = document.getElementById('qrcode-canvas');

        const generateQrCode = () => {
            qrcodeCanvasDiv.innerHTML = '';
            
            try {
                if (typeof qrcode !== 'undefined') {
                    // qrcode(typeNumber, errorCorrectionLevel)
                    const qr = qrcode(0, 'H'); 
                    qr.addData(qrDataToEncode);
                    qr.make();
                    
                    // The function definition for create Canvas is included in the self-contained script block
                    const canvas = qr.createCanvas(4); 
                    qrcodeCanvasDiv.appendChild(canvas);
                    
                    canvas.style.display = 'block';
                    canvas.style.margin = '0 auto';

                } else {
                    qrcodeCanvasDiv.innerHTML = '<p class="text-red-500">تعذر تحميل أداة رمز QR محلياً. يرجى تحديث الصفحة.</p>';
                }
            } catch (e) {
                console.error("QR Code generation failed:", e);
                qrcodeCanvasDiv.innerHTML = '<p class="text-red-500">حدث خطأ أثناء إنشاء رمز QR. يرجى محاولة فتح الرابط مباشرة.</p>';
            }
        };


        // --- Event Listeners ---
        if (chatForm) {
            chatForm.addEventListener('submit', (e) => {
                e.preventDefault();
                const query = userInput.value;
                if (query && !sendButton.disabled) {
                    sendMessage(query);
                }
            });
        }

        // QR Button Listener
        qrButton.addEventListener('click', () => {
            qrModalOverlay.style.display = 'flex';
            // Generate QR code immediately upon click
            generateQrCode(); 
        });


        closeQrModal.addEventListener('click', () => {
            qrModalOverlay.style.display = 'none';
        });

        qrModalOverlay.addEventListener('click', (e) => {
            if (e.target === qrModalOverlay) {
                qrModalOverlay.style.display = 'none';
            }
        });

        // Enable button when input is not empty
        userInput.addEventListener('input', () => {
            sendButton.disabled = userInput.value.trim() === '' || !isAuthReady;
        });

    </script>
